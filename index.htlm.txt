      <div class="option" data-idx="${idx}" data-ans="A">A：马上冲！热闹怎么能少了我</div>

// 选项点击
const answers = {};
document.querySelectorAll('.option').forEach(opt => {
  opt.addEventListener('click', () => {
    const idx = opt.dataset.idx;
    const ans = opt.dataset.ans;
    answers[idx] = ans;
    opt.parentElement.querySelectorAll('.option').forEach(o => o.classList.remove('selected'));
    opt.classList.add('selected');
  });
});

// 提交计算
document.getElementById('submitBtn').addEventListener('click', () => {
  let count = {F:0,K:0,T:0,I:0};
  for(let idx in answers){
    const q = questions[idx];
    const ans = answers[idx];
    const c = q[ans];
    if(c) count[c]++;
  }

  // 按最高排序组合
  const keys = Object.keys(count).sort((a,b)=>count[b]-count[a]);
  const typeKey = keys.join('');
  const res = types[typeKey] || types.FKTI;

  document.getElementById('typeName').innerText = res.name;
  document.getElementById('typeCode').innerText = res.code;
  document.getElementById('typeFeature').innerText = res.feature;
  document.getElementById('typeJob').innerText = res.job;
  document.getElementById('typeLove').innerText = res.love;

  document.getElementById('quiz').style.display = 'none';
  document.getElementById('submitBtn').style.display = 'none';
  document.getElementById('result').style.display = 'block';
});
